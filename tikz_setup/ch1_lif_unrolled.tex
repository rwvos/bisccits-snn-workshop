% F3 - the single LIF neuron (exemplar) UNROLLED over 3 timesteps (t-1, t, t+1).
% One `lifnode` copy per timestep; membrane state is carried forward along the
% top via `leak` arrows labelled beta. Bridges the folded recurrence (F1) to the
% explicit time-loop the learner implements.
\documentclass[border=2pt]{standalone}
\usepackage{tikz}
\input{style.tex}

\begin{document}
\begin{tikzpicture}[x=1cm, y=1cm, node distance=18mm]

  % --- one copy of the SAME neuron per timestep ---
  \node[lifnode] (nm) {$\Vmem[t\!-\!1]$};
  \node[lifnode, right=18mm of nm] (n0) {$\Vmem[t]$};
  \node[lifnode, right=18mm of n0] (np) {$\Vmem[t\!+\!1]$};

  % --- continuation markers (the neuron lived before and lives after) ---
  \node[iolabel, text=tabgray, left=7mm of nm]  (lc) {$\cdots$};
  \node[iolabel, text=tabgray, right=7mm of np] (rc) {$\cdots$};

  % --- membrane state carried forward in time along the top (leak, beta) ---
  \draw[leak] (lc)      -- (nm.west);
  \draw[leak] (nm.east) -- (n0.west) node[midway, above, iolabel, text=tabgray] {$\bbeta$};
  \draw[leak] (n0.east) -- (np.west) node[midway, above, iolabel, text=tabgray] {$\bbeta$};
  \draw[leak] (np.east) -- (rc);

  % --- input current into each node (from below) ---
  \draw[current] ($(nm.south)+(0,-1.1)$) -- (nm.south)
        node[midway, right, iolabel, text=tabblue] {$\Iin[t\!-\!1]$};
  \draw[current] ($(n0.south)+(0,-1.1)$) -- (n0.south)
        node[midway, right, iolabel, text=tabblue] {$\Iin[t]$};
  \draw[current] ($(np.south)+(0,-1.1)$) -- (np.south)
        node[midway, right, iolabel, text=tabblue] {$\Iin[t\!+\!1]$};

  % --- spike output out of each node (upward) ---
  \draw[spike] (nm.north) -- ($(nm.north)+(0,1.1)$)
        node[midway, right, iolabel] {$\Sout[t\!-\!1]$};
  \draw[spike] (n0.north) -- ($(n0.north)+(0,1.1)$)
        node[midway, right, iolabel] {$\Sout[t]$};
  \draw[spike] (np.north) -- ($(np.north)+(0,1.1)$)
        node[midway, right, iolabel] {$\Sout[t\!+\!1]$};

  % --- caption ---
  \node[fignote, below=14mm of n0, text width=9.5cm, align=center]
       {the same neuron, one copy per timestep; membrane state flows along $\bbeta$};
\end{tikzpicture}
\end{document}
