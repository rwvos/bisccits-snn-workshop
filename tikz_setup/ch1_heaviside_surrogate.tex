% F4 - WHY we cannot backprop the spike, and what we swap in.
% Conceptual 3-panel schematic (NOT a data plot), left to right:
%   (1) spike = Heaviside step, (2) its true derivative = Dirac delta (unusable),
%   (3) the smooth surrogate derivative we substitute.
% Each panel: small L-shaped axis + red dashed threshold line at V_thr.
\documentclass[border=2pt]{standalone}
\usepackage{tikz}
\input{style.tex}

\begin{document}
\begin{tikzpicture}[x=1cm, y=1cm]

  % a reusable panel: draws the L-axes and the dashed threshold at x=0 (the threshold).
  % #1 = x-shift of the panel origin, #2 = panel title (above).
  \newcommand{\panel}[2]{
    \begin{scope}[shift={(#1,0)}]
      % L-shaped axes
      \draw[inkblack, line width=0.7pt] (-1.2,0) -- (1.2,0);   % horizontal (V)
      \draw[inkblack, line width=0.7pt] (-1.2,0) -- (-1.2,2.0); % vertical
      \node[iolabel, text=inkblack] at (1.45,0) {$\Vmem$};
      % threshold line (red dashed) centred at x=0
      \draw[thresholdline] (0,0) -- (0,1.95);
      \node[iolabel, text=tabred] at (0,2.2) {$\Vthr$};
      % panel title
      \node[align=center, font=\footnotesize] at (0,-0.85) {#2};
    \end{scope}}

  % --- Panel 1: Heaviside step (the spike itself) ---
  \panel{0}{spike: Heaviside\\ $\Sout=H(\Vmem-\Vthr)$}
  \begin{scope}[shift={(0,0)}]
    \draw[inkblack, line width=1.1pt] (-1.2,0.15) -- (0,0.15);   % low (0)
    \draw[inkblack, line width=1.1pt] (0,1.55) -- (1.2,1.55);    % high (1)
    \draw[inkblack, line width=1.1pt] (0,0.15) -- (0,1.55);      % the jump
    \node[iolabel, text=inkblack] at (-0.85,0.42) {$0$};
    \node[iolabel, text=inkblack] at ( 0.85,1.82) {$1$};
  \end{scope}

  % --- Panel 2: true derivative = Dirac delta (unusable) ---
  \panel{4}{true derivative: Dirac $\delta$\\ (unusable)}
  \begin{scope}[shift={(4,0)}]
    \draw[inkblack, line width=1.1pt] (-1.2,0.15) -- (1.2,0.15); % 0 everywhere
    % single tall impulse arrow AT the threshold; the red dashed V_thr line
    % (drawn by \panel) runs taller behind it and stays visible to the side.
    \draw[spike] (0,0.15) -- (0,1.6) node[right=1pt, iolabel, text=inkblack] {$\infty$};
    \node[fignote, align=center, text width=3.4cm] at (0,-1.55)
         {$0$ everywhere, $\infty$ at threshold --- no gradient};
  \end{scope}

  % --- Panel 3: smooth surrogate derivative we substitute ---
  \panel{8}{surrogate:\\ $\sigma'\!\big(\slope\,(\Vmem-\Vthr)\big)$}
  \begin{scope}[shift={(8,0)}]
    % smooth symmetric bump centred on the threshold (stylised)
    \draw[tabpurple, line width=1.1pt]
        (-1.2,0.18)
        .. controls (-0.45,0.22) and (-0.30,1.55) .. (0,1.55)
        .. controls ( 0.30,1.55) and ( 0.45,0.22) .. (1.2,0.18);
  \end{scope}

\end{tikzpicture}
\end{document}
